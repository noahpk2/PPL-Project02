public class Main {
    
    public static main(String[] args) 
        int x = 20;
        int y = 18
        if (x > y) {
          System.out.println("x is greater than y");
        }  
      }
    
    public static void main(String[] args) {
        for (int i = 0; i < 5; i++) {
          System.out.println(i);
          
      }

    static void myMethod() {
      System.out.println("This is a method!");
  }

  public static void main(String[] args) {
    myMethod();
  }
    
}
